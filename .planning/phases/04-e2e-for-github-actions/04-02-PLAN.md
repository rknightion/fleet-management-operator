Create test/e2e/pipeline_test.go with Ginkgo E2E tests for Pipeline CR lifecycle.

File must have:
- Build tag: //go:build e2e
- Package: e2e
- Imports: ginkgo/v2, gomega, os/exec, time, fmt, and test utils

Use the same namespace constant from e2e_test.go. Reference fixtures by relative path
from the project root (test/e2e/fixtures/*.yaml).

Test scenarios using Ginkgo Describe/Context/It pattern:

Context("Pipeline Lifecycle", Ordered, func() {

  It("should create an Alloy pipeline and reach Ready state", func() {
    By("applying the valid Alloy pipeline fixture")
    kubectl apply -f test/e2e/fixtures/valid-alloy-pipeline.yaml -n {namespace}

    By("waiting for the pipeline to have a Fleet Management ID")
    Eventually (2min, 5s): kubectl get pipeline test-alloy-pipeline -n {namespace}
      -o jsonpath={.status.id} should NotBeEmpty()

    By("verifying the Ready condition is True")
    Eventually (2min, 5s): kubectl get pipeline test-alloy-pipeline -n {namespace}
      -o jsonpath={.status.conditions[?(@.type=='Ready')].status} should Equal("True")

    By("verifying observedGeneration matches generation")
    Get generation and observedGeneration via jsonpath, compare they are equal and > 0
  })

  It("should create an OpenTelemetry Collector pipeline and reach Ready state", func() {
    By("applying the valid OTEL pipeline fixture")
    kubectl apply -f test/e2e/fixtures/valid-otel-pipeline.yaml -n {namespace}

    By("waiting for the pipeline to reach Ready state")
    Eventually (2min, 5s): Ready condition status == "True"

    By("verifying the Fleet Management ID is assigned")
    status.id should NotBeEmpty()
  })

  It("should update a pipeline and reconcile the new spec", func() {
    By("applying the updated Alloy pipeline fixture")
    kubectl apply -f test/e2e/fixtures/update-alloy-pipeline.yaml -n {namespace}

    By("waiting for observedGeneration to advance")
    Get current generation first, then Eventually check that observedGeneration >= that generation

    By("verifying the pipeline remains Ready after update")
    Eventually (2min, 5s): Ready condition status == "True"
  })

  It("should reject an invalid pipeline via webhook", func() {
    By("attempting to apply a pipeline with mismatched configType")
    cmd := exec.Command("kubectl", "apply", "-f",
      "test/e2e/fixtures/invalid-mismatch-pipeline.yaml", "-n", namespace)
    output, err := utils.Run(cmd)
    // Webhook should reject it - expect an error
    Expect(err).To(HaveOccurred(), "Webhook should reject mismatched configType")
    Expect(output).To(ContainSubstring("denied"), "Error should indicate webhook denial")
    // NOTE: If webhooks are not enabled in the E2E environment (no cert-manager),
    // this test should be skipped. Check if CERT_MANAGER_INSTALL_SKIP is set.
    // Use: if os.Getenv("CERT_MANAGER_INSTALL_SKIP") == "true" { Skip("...") }
  })

  It("should delete a pipeline and remove the finalizer", func() {
    By("deleting the Alloy pipeline")
    kubectl delete pipeline test-alloy-pipeline -n {namespace}

    By("verifying the pipeline is fully deleted")
    Eventually (2min, 5s): kubectl get pipeline test-alloy-pipeline -n {namespace}
      should fail (NotFound error in output)

    By("deleting the OTEL pipeline")
    kubectl delete pipeline test-otel-pipeline -n {namespace}

    By("verifying the OTEL pipeline is fully deleted")
    Eventually (2min, 5s): kubectl get pipeline test-otel-pipeline -n {namespace}
      should fail (NotFound error in output)
  })
})

IMPORTANT patterns:
- Use Ordered to ensure tests run in sequence (create before update before delete)
- Use Eventually with Gomega's async assertion pattern (func(g Gomega) { ... })
- Set timeout 2*time.Minute and polling 5*time.Second for all Eventually calls
- Use exec.Command + utils.Run for all kubectl interactions (consistent with existing tests)
- Use DeferCleanup in case tests fail partway through to clean up created resources
- All kubectl commands should specify -n {namespace}
